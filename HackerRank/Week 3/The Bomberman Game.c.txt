The Bomberman Game.c

#include <stdio.h>

#define MAX 200

int R, C, N;
char grid[MAX][MAX];
char temp[MAX][MAX];
void copyGrid(char dest[MAX][MAX], char src[MAX][MAX]) {
    for (int i = 0; i < R; i++)
        for (int j = 0; j < C; j++)
            dest[i][j] = src[i][j];
}
void fillWithBombs(char g[MAX][MAX]) {
    for (int i = 0; i < R; i++)
        for (int j = 0; j < C; j++)
            g[i][j] = 'O';
}
void detonate(char g[MAX][MAX], char ref[MAX][MAX]) {
    int dr[] = {-1, 1, 0, 0};
    int dc[] = {0, 0, -1, 1};
    
    fillWithBombs(g);
    
    for (int i = 0; i < R; i++) {
        for (int j = 0; j < C; j++) {
            if (ref[i][j] == 'O') {
                g[i][j] = '.';
                for (int k = 0; k < 4; k++) {
                    int nr = i + dr[k];
                    int nc = j + dc[k];
                    if (nr >= 0 && nr < R && nc >= 0 && nc < C)
                        g[nr][nc] = '.';
                }
            }
        }
    }
}

int main() {
    scanf("%d %d %d", &R, &C, &N);
    for (int i = 0; i < R; i++)
        scanf("%s", grid[i]);
    
    if (N == 1) {
    } else if (N % 2 == 0) {
        fillWithBombs(grid);
    } else {
        char first[MAX][MAX], second[MAX][MAX];
        detonate(first, grid);       
        detonate(second, first);     
        if ((N - 3) % 4 == 0)
            copyGrid(grid, first);
        else
            copyGrid(grid, second);
    }
    
    for (int i = 0; i < R; i++) {
        for (int j = 0; j < C; j++)
            printf("%c", grid[i][j]);
        printf("\n");
    }
    
    return 0;
}
