int max(int a, int b) {
    return (a > b) ? a : b;
}

int flippingMatrix(int n, int matrix[256][256]) {
    int sum = 0;
    int size = 2 * n;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            int val1 = matrix[i][j];
            int val2 = matrix[i][size - j - 1];
            int val3 = matrix[size - i - 1][j];
            int val4 = matrix[size - i - 1][size - j - 1];
            int maxVal = max(max(val1, val2), max(val3, val4));
            sum += maxVal;
            }
            }
            return sum;
            }
