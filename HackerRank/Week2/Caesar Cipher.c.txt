Caesar Cipher.c

#include <stdio.h>
#include <string.h>
#include <ctype.h>

void caesarCipher(char *s, int k) {
    int len = strlen(s);
    k = k % 26;
    for (int i = 0; i < len; i++) {
        if (isalpha(s[i])) {
            char base = islower(s[i]) ? 'a' : 'A';
            s[i] = (s[i] - base + k) % 26 + base;
        }
    }
}

int main() {
    int n, k;
    char s[200];
    scanf("%d", &n);
    scanf("%s", s); 
    scanf("%d", &k);
    caesarCipher(s, k);
    printf("%s\n", s);
    return 0;
}
