#include <stdio.h>
#include <stdbool.h>
#include <stdint.h>

bool isPowerOfTwo(unsigned long long n) {
    return (n && !(n & (n - 1)));
}
unsigned long long largestPowerOfTwo(unsigned long long n) {
    unsigned long long p = 1;
    while (p << 1ULL < n) {
        p <<= 1ULL;
    }
    return p;
}

void counterGame(unsigned long long n) {
    int turns = 0;

    while (n > 1) {
        if (isPowerOfTwo(n)) {
            n /= 2;
        } else {
            n -= largestPowerOfTwo(n);
        }
        turns++;
    }
    if (turns % 2 == 1) {
        printf("Louise\n");
    } else {
        printf("Richard\n");
    }
}

int main() {
    int t;
    unsigned long long n;

    scanf("%d", &t);
    while (t--) {
        scanf("%llu", &n);
        counterGame(n);
    }

    return 0;
}
