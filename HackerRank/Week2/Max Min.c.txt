#include <stdio.h>
#include <stdlib.h>

int compare(const void *a, const void *b) {
    return (*(int*)a - *(int*)b);
}

int min_unfairness(int arr[], int n, int k) {
    qsort(arr, n, sizeof(int), compare);
    int min_unfairness = arr[k-1] - arr[0];
    for (int i = 1; i <= n - k; i++) {
        int unfairness = arr[i+k-1] - arr[i];
        if (unfairness < min_unfairness) {
            min_unfairness = unfairness;
        }
    }
    return min_unfairness;
}

int main() {
    int n, k;
    scanf("%d %d", &n, &k);
    int arr[n];
    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }
    printf("%d\n", min_unfairness(arr, n, k));
    return 0;
}
